          <button class="qa" style="color:var(--color-text-info);" onclick="updateStatus('${a.id}','shortlisted')"><i class="ti ti-star" style="font-size:13px;" aria-hidden="true"></i>Shortlist</button>
          <button class="qa" onclick="updateStatus('${a.id}','incomplete')"><i class="ti ti-alert-triangle" style="font-size:13px;" aria-hidden="true"></i>Request documents</button>
          <button class="qa" style="color:var(--color-text-danger);" onclick="updateStatus('${a.id}','rejected')"><i class="ti ti-thumb-down" style="font-size:13px;" aria-hidden="true"></i>Reject application</button>
        </div>
      </div>
    </div>
  </td></tr>`;
}

function renderTable(){
  const f=getFiltered();
  const tbody=document.getElementById("tbody");
  const empty=document.getElementById("empty");
  const cnt=document.getElementById("rcnt");
  if(cnt)cnt.textContent=f.length;
  if(!f.length){tbody.innerHTML="";empty.style.display="block";return;}
  empty.style.display="none";
  const bg0="var(--color-background-primary)";
  const bg1="var(--color-background-secondary)";
  tbody.innerHTML=f.map((a,i)=>{
    const exp=ST.expanded===a.id;
    const rbg=exp?bg1:(i%2===0?bg0:bg1);
    const ini=initials(a.name);
    const uc=a.utme>=250?"var(--color-text-success)":a.utme>=220?"var(--color-text-primary)":"var(--color-text-danger)";
    const entC=a.entry==="Direct Entry"?"var(--color-background-info)":"var(--color-background-tertiary)";
    const entT=a.entry==="Direct Entry"?"var(--color-text-info)":"var(--color-text-secondary)";
    const entS=a.entry==="Direct Entry"?"DE":"UTME";
    const mr=`<tr class="drow" onclick="toggleExp('${a.id}')">
      <td style="padding:10px 12px;background:${rbg};">
        <div style="display:flex;align-items:center;gap:8px;">
          <div style="width:28px;height:28px;border-radius:50%;background:var(--color-background-info);display:flex;align-items:center;justify-content:center;font-size:10px;font-weight:500;color:var(--color-text-info);flex-shrink:0;">${ini}</div>
          <div style="min-width:0;">
            <div style="font-size:13px;font-weight:500;color:var(--color-text-primary);overflow:hidden;text-overflow:ellipsis;white-space:nowrap;">${a.name}</div>
            <div style="font-size:10px;color:var(--color-text-tertiary);">${a.id}</div>
          </div>
        </div>
      </td>
      <td style="padding:10px 12px;background:${rbg};">
        <div style="font-size:12px;color:var(--color-text-primary);overflow:hidden;text-overflow:ellipsis;white-space:nowrap;">${a.prog}</div>
        <span style="font-size:10px;padding:1px 5px;border-radius:4px;background:${entC};color:${entT};">${entS}</span>
      </td>
      <td style="padding:10px 12px;background:${rbg};">
        <div style="font-size:14px;font-weight:500;color:${uc};">${a.utme}</div>
        <div style="font-size:10px;color:var(--color-text-tertiary);">/400</div>
      </td>
      <td style="padding:10px 12px;background:${rbg};">${docsBar(a.docs)}</td>
      <td style="padding:10px 12px;background:${rbg};">${badge(a.status)}</td>
      <td style="padding:10px 4px;background:${rbg};text-align:center;">
        <i class="ti ${exp?"ti-chevron-up":"ti-chevron-down"}" style="font-size:14px;color:var(--color-text-tertiary);" aria-hidden="true"></i>
      </td>
    </tr>`;
    return mr+(exp?expandedRow(a):"");
  }).join("");
}

function renderProgFilters(){
  document.getElementById("prog-filters").innerHTML=PROGS.map(p=>{
    const act=ST.prog===p?"active":"";
    return `<div class="prog-item ${act}" onclick="setProg('${p}')"><span style="overflow:hidden;text-overflow:ellipsis;white-space:nowrap;">${p==="All"?"All programmes":p}</span><span class="cpill">${pcnt(p)}</span></div>`;
  }).join("");
}

function render(){renderProgFilters();renderTable();}
function toggleExp(id){ST.expanded=ST.expanded===id?null:id;renderTable();}
function setProg(p){ST.prog=p;render();}
function onSearch(v){ST.search=v;renderTable();}
function onStatusFilter(v){ST.status=v;renderTable();}

render();
</script>
