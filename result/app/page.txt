import type React from "react"
import { SavedResultsView } from "@/components/saved-results-view" // Declare the variable before using it

interface Course {
  id: string
  code: string
  unit: number
  score: number
  gradePoint: number
}

interface StudentInfo {
  name: string
  matricNumber: string
  level: string
  faculty: string
  department: string
  academicSession: string
  semester: string
}

interface ExcelStudent {
  sn: number
  matricNumber: string
  name: string
  courses: { code: string; score: number; unit: number }[]
}

interface ExcelData {
  metadata: Partial<StudentInfo>
  students: ExcelStudent[]
}

function getGradePoint(score: number): number {
  if (score >= 70) return 5
  if (score >= 60) return 4
  if (score >= 50) return 3
  if (score >= 45) return 2
  if (score >= 40) return 1
  return 0
}

function parseExcelSheet(sheet: any): ExcelData {
  const data = XLSX.utils.sheet_to_json(sheet, { header: 1, defval: "" }) as any[][]

  const result: ExcelData = {
    metadata: {},
    students: [],
  }

  /* ================= METADATA EXTRACTION ================= */

  for (let i = 0; i < Math.min(15, data.length); i++) {
    const row = data[i]
    if (!row) continue

    const rowText = row.join(" ").toUpperCase()

    if (rowText.includes("LEVEL")) {
      const match = rowText.match(/LEVEL\s*:\s*(\d+)/i)
      if (match) result.metadata.level = match[1]
    }

    if (rowText.includes("FACULTY OF")) {
      const match = rowText.match(/FACULTY\s+OF\s+(.+)/i)
      if (match) result.metadata.faculty = match[1].trim()
    }

    if (rowText.includes("DEPARTMENT OF")) {
      const match = rowText.match(/DEPARTMENT\s+OF\s+(.+)/i)
      if (match) result.metadata.department = match[1].trim()
    }

    if (rowText.includes("ACADEMIC SESSION")) {
      const match = rowText.match(/(\d{4}\/\d{4})/)
      if (match) result.metadata.academicSession = match[1]
    }

    if (rowText.includes("SEMESTER")) {
      const match = rowText.match(/(FIRST|SECOND|THIRD)/i)
      if (match) result.metadata.semester = match[1]
    }
  }

  /* ================= HEADER ROW DETECTION ================= */

  let headerRowIndex = -1
  let snCol = -1
  let matricCol = -1
  let nameCol = -1
  let courseStartCol = -1

  for (let i = 0; i < data.length; i++) {
    const row = data[i]
    if (!row) continue

    for (let j = 0; j < row.length; j++) {
      const cell = String(row[j] || "")
        .toLowerCase()
        .trim()

      if (cell === "s/n" || cell === "sn") {
        snCol = j
        headerRowIndex = i
      } else if (cell.includes("matric") && headerRowIndex === i) {
        matricCol = j
      } else if (cell === "name" && headerRowIndex === i) {
        nameCol = j
        courseStartCol = j + 1
      }
    }

    if (headerRowIndex !== -1 && snCol !== -1 && matricCol !== -1 && nameCol !== -1) break
  }

  if (headerRowIndex === -1) return result

  /* ================= COURSE CODE + UNIT EXTRACTION (FIXED) ================= */

  const courseCodes: string[] = []
  const courseUnits: number[] = []

  for (let j = courseStartCol; j < data[headerRowIndex].length; j++) {
    const raw = String(data[headerRowIndex][j] || "")
      .trim()
      .toUpperCase()

    if (!raw || raw === "TOTAL" || raw === "GRADE") break

    let cleanCode = raw
    let unit = 0

    // Extract unit from brackets e.g. CSC101 (3)
    const unitMatch = raw.match(/$$(\d+)$$/)
    if (unitMatch) {
      unit = Number.parseInt(unitMatch[1], 10)
      cleanCode = raw.replace(/\s*$$\d+$$/, "").trim()
    }

    courseCodes.push(cleanCode)
    courseUnits.push(unit)
  }

  /* ================= STUDENT DATA EXTRACTION ================= */

  for (let i = headerRowIndex + 1; i < data.length; i++) {
    const row = data[i]
    if (!row) continue

    const sn = Number(row[snCol])
    if (isNaN(sn)) continue

    const student: ExcelStudent = {
      sn,
      matricNumber: String(row[matricCol] || "").trim(),
      name: String(row[nameCol] || "").trim(),
      courses: [],
    }

    for (let j = 0; j < courseCodes.length; j++) {
      const scoreCol = courseStartCol + j
      let scoreText = String(row[scoreCol] || "").trim()

      scoreText = scoreText.replace(/\s*[A-Za-z]+\s*/, "")
      const scoreMatch = scoreText.match(/\d+/)
      const score = scoreMatch ? Number(scoreMatch[0]) : 0

      student.courses.push({
        code: courseCodes[j],
        unit: courseUnits[j],
        score,
      })
    }

    result.students.push(student)
  }

  return result
}

export default function ResultCalculator() {
  const [studentInfo, setStudentInfo] = useState<StudentInfo>({
    name: "",
    matricNumber: "",
    level: "",
    faculty: "",
    department: "",
    academicSession: "",
    semester: "",
  })

  const [courses, setCourses] = useState<Course[]>([])
  const [currentCourse, setCurrentCourse] = useState({ code: "", unit: "", score: "" })
  const [showResult, setShowResult] = useState(false)
  const [selectedSheet, setSelectedSheet] = useState<string>("")
  const [availableSheets, setAvailableSheets] = useState<string[]>([])
  const [workbook, setWorkbook] = useState<XLSX.WorkBook | null>(null)

  const [excelData, setExcelData] = useState<ExcelData | null>(null)
  const [selectedStudentIndex, setSelectedStudentIndex] = useState<number>(-1)
  const [view, setView] = useState<"calculator" | "saved">("calculator")

  const handleFileUpload = (event: React.ChangeEvent<HTMLInputElement>) => {
    const file = event.target.files?.[0]
    if (!file) return

    const reader = new FileReader()
    reader.onload = (e) => {
      const data = new Uint8Array(e.target?.result as ArrayBuffer)
      const wb = XLSX.read(data, { type: "array" })
      setWorkbook(wb)
      setAvailableSheets(wb.SheetNames)

      if (wb.SheetNames.length > 0) {
        handleSheetSelection(wb, wb.SheetNames[0])
      }
    }
    reader.readAsArrayBuffer(file)
  }

  const handleSheetSelection = (wb: XLSX.WorkBook, sheetName: string) => {
    setSelectedSheet(sheetName)
    const sheet = wb.Sheets[sheetName]

    const parsed = parseExcelSheet(sheet)
    setExcelData(parsed)

    const cleanedDepartment = sheetName.replace(/\s*$$\d+$$\s*$/, "").trim() || "-"

    setStudentInfo({
      name: "",
      matricNumber: "",
      level: parsed.metadata.level || "-",
      faculty: parsed.metadata.faculty || "-",
      department: cleanedDepartment,
      academicSession: parsed.metadata.academicSession || "-",
      semester: parsed.metadata.semester || "-",
    })

    // Reset student selection
    setSelectedStudentIndex(-1)
    setCourses([])
  }

  const handleStudentSelection = (studentNum: number) => {
    if (!excelData || studentNum < 1 || studentNum > excelData.students.length) return

    const student = excelData.students[studentNum - 1]
    setSelectedStudentIndex(studentNum - 1)

    // Update student info
    setStudentInfo((prev) => ({
      ...prev,
      name: student.name,
      matricNumber: student.matricNumber,
    }))

    // Convert courses to Course format with default unit of 3
    const loadedCourses: Course[] = student.courses.map((course, index) => ({
      id: Date.now().toString() + index,
      code: course.code,
      unit: course.unit, // Use extracted unit from Excel
      score: course.score,
      gradePoint: getGradePoint(course.score),
    }))

    setCourses(loadedCourses)
  }

  const updateCourseUnit = (id: string, newUnit: number) => {
    setCourses((prevCourses) => prevCourses.map((course) => (course.id === id ? { ...course, unit: newUnit } : course)))
  }

  const addCourse = () => {
    if (!currentCourse.code || !currentCourse.unit || !currentCourse.score) {
      alert("Please fill in all course details")
      return
    }

    const unit = Number.parseInt(currentCourse.unit)
    const score = Number.parseInt(currentCourse.score)

    if (unit < 1 || unit > 6) {
      alert("Unit must be between 1 and 6")
      return
    }

    if (score < 0 || score > 100) {
      alert("Score must be between 0 and 100")
      return
    }

    const gradePoint = getGradePoint(score)

    const newCourse: Course = {
      id: Date.now().toString(),
      code: currentCourse.code.toUpperCase(),
      unit,
      score,
      gradePoint,
    }

    setCourses([...courses, newCourse])
    setCurrentCourse({ code: "", unit: "", score: "" })
  }

  const removeCourse = (id: string) => {
    setCourses(courses.filter((c) => c.id !== id))
  }

  const calculateResults = () => {
    if (!studentInfo.name.trim()) {
      alert("Please enter candidate name")
      return
    }

    if (courses.length === 0) {
      alert("Please add at least one course")
      return
    }

    setShowResult(true)
  }

  const resetForm = () => {
    setStudentInfo({
      name: "",
      matricNumber: "",
      level: "",
      faculty: "",
      department: "",
      academicSession: "",
      semester: "",
    })
    setCourses([])
    setCurrentCourse({ code: "", unit: "", score: "" })
    setShowResult(false)
    setSelectedSheet("")
    setAvailableSheets([])
    setWorkbook(null)
    setExcelData(null)
    setSelectedStudentIndex(-1)
  }

  const handleViewSaved = () => {
    setView("saved")
  }

  const handleBackToCalculator = () => {
    setView("calculator")
  }

  const totalUnits = courses.reduce((sum, course) => sum + course.unit, 0)
  const totalUnitsPassed = courses.reduce((sum, course) => {
    return course.gradePoint === 0 ? sum : sum + course.unit
  }, 0)
  const totalWGP = courses.reduce((sum, course) => sum + course.unit * course.gradePoint, 0)
  const cgpa = totalUnits > 0 ? (totalWGP / totalUnits).toFixed(2) : "0.00"

  return (
    <div className="min-h-screen bg-gradient-to-br from-blue-50 via-white to-slate-50 py-8 px-4">
      <div className="max-w-7xl mx-auto">
        {view === "saved" ? (
          <SavedResultsView onBack={handleBackToCalculator} />
        ) : (
          <>
            <div className="text-center mb-8">
              <h1 className="text-4xl font-bold text-slate-900 mb-2">Precious Cornerstone University</h1>
              <p className="text-slate-600 text-lg">Student Result Calculator</p>
              <div className="mt-4">
                <Button
                  onClick={handleViewSaved}
                  variant="outline"
                  className="border-blue-300 hover:bg-blue-50 bg-transparent"
                >
                  View Saved Results
                </Button>
              </div>
            </div>

            {!showResult ? (
              <div className="grid lg:grid-cols-2 gap-6">
                {/* Input Form */}
                <Card className="border-blue-200 shadow-lg">
                  <CardHeader className="bg-gradient-to-r from-blue-600 to-blue-700 text-white">
                    <CardTitle className="text-2xl">Enter Student Details</CardTitle>
                    <CardDescription className="text-blue-100">Add candidate information and courses</CardDescription>
                  </CardHeader>
                  <CardContent className="pt-6 space-y-6">
                    {/* Excel Upload */}
                    <div className="space-y-2">
                      <Label htmlFor="excel" className="text-slate-700 font-semibold">
                        Upload Excel File (Optional)
                      </Label>
                      <div className="flex gap-2">
                        <Input
                          id="excel"
                          type="file"
                          accept=".xlsx,.xls"
                          onChange={handleFileUpload}
                          className="border-slate-300"
                        />
                        <Button variant="outline" className="shrink-0 bg-transparent">
                          <Upload className="h-4 w-4" />
                        </Button>
                      </div>

                      {availableSheets.length > 0 && (
                        <div className="space-y-2">
                          <Label className="text-slate-700">Select Department Sheet</Label>
                          <select
                            className="w-full p-2 border border-slate-300 rounded-md"
                            value={selectedSheet}
                            onChange={(e) => workbook && handleSheetSelection(workbook, e.target.value)}
                          >
                            {availableSheets.map((sheet) => (
                              <option key={sheet} value={sheet}>
                                {sheet}
                              </option>
                            ))}
                          </select>
                        </div>
                      )}

                      {excelData && excelData.students.length > 0 && (
                        <div className="space-y-2 p-4 bg-green-50 border border-green-200 rounded-md">
                          <p className="text-green-900 font-semibold">
                            ✓ Found {excelData.students.length} student{excelData.students.length !== 1 ? "s" : ""} in
                            this sheet
                          </p>
                          <div className="space-y-2">
                            <Label className="text-slate-700">Select Student Number (S/N)</Label>
                            <select
                              className="w-full p-2 border border-slate-300 rounded-md"
                              value={selectedStudentIndex + 1}
                              onChange={(e) => handleStudentSelection(Number(e.target.value))}
                            >
                              <option value={0}>-- Select a student --</option>
                              {excelData.students.map((student) => (
                                <option key={student.sn} value={student.sn}>
                                  {student.sn}. {student.name} ({student.matricNumber})
                                </option>
                              ))}
                            </select>
                          </div>
                        </div>
                      )}
                    </div>

                    {/* New student info fields */}
                    <div className="grid grid-cols-2 gap-4">
                      <div className="space-y-2">
                        <Label htmlFor="matric" className="text-slate-700">
                          Matric Number
                        </Label>
                        <Input
                          id="matric"
                          placeholder="e.g., PCU/2023/001"
                          value={studentInfo.matricNumber}
                          onChange={(e) => setStudentInfo({ ...studentInfo, matricNumber: e.target.value })}
                          className="border-slate-300"
                        />
                      </div>

                      <div className="space-y-2">
                        <Label htmlFor="level" className="text-slate-700">
                          Level
                        </Label>
                        <Input
                          id="level"
                          placeholder="e.g., 100, 200"
                          value={studentInfo.level}
                          onChange={(e) => setStudentInfo({ ...studentInfo, level: e.target.value })}
                          className="border-slate-300"
                        />
                      </div>
                    </div>

                    <div className="space-y-2">
                      <Label htmlFor="faculty" className="text-slate-700">
                        Faculty
                      </Label>
                      <Input
                        id="faculty"
                        placeholder="e.g., Science"
                        value={studentInfo.faculty}
                        onChange={(e) => setStudentInfo({ ...studentInfo, faculty: e.target.value })}
                        className="border-slate-300"
                      />
                    </div>

                    <div className="space-y-2">
                      <Label htmlFor="department" className="text-slate-700">
                        Department
                      </Label>
                      <Input
                        id="department"
                        placeholder="e.g., Computer Science"
                        value={studentInfo.department}
                        onChange={(e) => setStudentInfo({ ...studentInfo, department: e.target.value })}
                        className="border-slate-300"
                      />
                    </div>

                    <div className="grid grid-cols-2 gap-4">
                      <div className="space-y-2">
                        <Label htmlFor="session" className="text-slate-700">
                          Academic Session
                        </Label>
                        <Input
                          id="session"
                          placeholder="e.g., 2023/2024"
                          value={studentInfo.academicSession}
                          onChange={(e) => setStudentInfo({ ...studentInfo, academicSession: e.target.value })}
                          className="border-slate-300"
                        />
                      </div>

                      <div className="space-y-2">
                        <Label htmlFor="semester" className="text-slate-700">
                          Semester
                        </Label>
                        <Input
                          id="semester"
                          placeholder="e.g., First, Second"
                          value={studentInfo.semester}
                          onChange={(e) => setStudentInfo({ ...studentInfo, semester: e.target.value })}
                          className="border-slate-300"
                        />
                      </div>
                    </div>

                    {/* Course Input */}
                    <div className="space-y-4 pt-4 border-t border-slate-200">
                      <h3 className="font-semibold text-slate-800">Add Course Manually</h3>

                      <div className="space-y-2">
                        <Label htmlFor="code" className="text-slate-700">
                          Course Code
                        </Label>
                        <Input
                          id="code"
                          placeholder="e.g., CYB101"
                          value={currentCourse.code}
                          onChange={(e) => setCurrentCourse({ ...currentCourse, code: e.target.value })}
                          className="border-slate-300"
                        />
                      </div>

                      <div className="grid grid-cols-2 gap-4">
                        <div className="space-y-2">
                          <Label htmlFor="unit" className="text-slate-700">
                            Unit
                          </Label>
                          <Input
                            id="unit"
                            type="number"
                            min="1"
                            max="6"
                            placeholder="1-6"
                            value={currentCourse.unit}
                            onChange={(e) => setCurrentCourse({ ...currentCourse, unit: e.target.value })}
                            className="border-slate-300"
                          />
                        </div>

                        <div className="space-y-2">
                          <Label htmlFor="score" className="text-slate-700">
                            Score
                          </Label>
                          <Input
                            id="score"
                            type="number"
                            min="0"
                            max="100"
                            placeholder="0-100"
                            value={currentCourse.score}
                            onChange={(e) => setCurrentCourse({ ...currentCourse, score: e.target.value })}
                            className="border-slate-300"
                          />
                        </div>
                      </div>

                      <Button onClick={addCourse} className="w-full bg-blue-600 hover:bg-blue-700">
                        <Plus className="h-4 w-4 mr-2" />
                        Add Course
                      </Button>
                    </div>

                    {courses.length > 0 && (
                      <Button
                        onClick={calculateResults}
                        className="w-full bg-green-600 hover:bg-green-700 text-lg py-6"
                      >
                        Calculate Results
                      </Button>
                    )}
                  </CardContent>
                </Card>

                {/* Courses List */}
                <Card className="border-slate-200 shadow-lg">
                  <CardHeader className="bg-slate-50 border-b border-slate-200">
                    <CardTitle className="text-slate-900">Courses Added ({courses.length})</CardTitle>
                    <CardDescription>Review and manage courses (units are editable)</CardDescription>
                  </CardHeader>
                  <CardContent className="pt-6">
                    {courses.length === 0 ? (
                      <div className="text-center py-12 text-slate-500">
                        <p className="text-lg">No courses added yet</p>
                        <p className="text-sm mt-2">Add courses manually or upload Excel file</p>
                      </div>
                    ) : (
                      <div className="space-y-3 max-h-[500px] overflow-y-auto">
                        {courses.map((course) => (
                          <div
                            key={course.id}
                            className="flex items-center justify-between p-4 bg-white border border-slate-200 rounded-lg hover:border-blue-300 transition-colors"
                          >
                            <div className="flex-1 grid grid-cols-4 gap-4 text-sm">
                              <div>
                                <span className="text-slate-500 text-xs block">Code</span>
                                <span className="font-semibold text-slate-900">{course.code}</span>
                              </div>
                              <div>
                                <span className="text-slate-500 text-xs block">Unit (editable)</span>
                                <Input
                                  type="number"
                                  min="1"
                                  max="6"
                                  value={course.unit}
                                  onChange={(e) => updateCourseUnit(course.id, Number(e.target.value))}
                                  className="h-7 w-16 text-sm font-semibold"
                                />
                              </div>
                              <div>
                                <span className="text-slate-500 text-xs block">Score</span>
                                <span className="font-semibold text-slate-900">{course.score}</span>
                              </div>
                              <div>
                                <span className="text-slate-500 text-xs block">GP</span>
                                <span className="font-semibold text-blue-600">{course.gradePoint}</span>
                              </div>
                            </div>
                            <Button
                              variant="ghost"
                              size="sm"
                              onClick={() => removeCourse(course.id)}
                              className="text-red-600 hover:text-red-700 hover:bg-red-50 ml-2"
                            >
                              <Trash2 className="h-4 w-4" />
                            </Button>
                          </div>
                        ))}
                      </div>
                    )}
                  </CardContent>
                </Card>
              </div>
            ) : (
              <ResultDisplay
                studentInfo={studentInfo}
                courses={courses}
                totalUnits={totalUnits}
                totalUnitsPassed={totalUnitsPassed}
                totalWGP={totalWGP}
                cgpa={cgpa}
                onReset={resetForm}
              />
            )}

            {/* Grade Scale Reference */}
            <Card className="mt-8 border-slate-200 bg-white/80 backdrop-blur">
              <CardHeader>
                <CardTitle className="text-slate-900 text-center">Grade Point Scale</CardTitle>
              </CardHeader>
              <CardContent>
                <div className="grid grid-cols-2 md:grid-cols-3 lg:grid-cols-6 gap-4 text-center">
                  <div className="p-3 bg-red-50 rounded-lg border border-red-200">
                    <div className="font-semibold text-red-900">0-39</div>
                    <div className="text-sm text-red-600">= 0 point</div>
                  </div>
                  <div className="p-3 bg-orange-50 rounded-lg border border-orange-200">
                    <div className="font-semibold text-orange-900">40-44</div>
                    <div className="text-sm text-orange-600">= 1 point</div>
                  </div>
                  <div className="p-3 bg-yellow-50 rounded-lg border border-yellow-200">
                    <div className="font-semibold text-yellow-900">45-49</div>
                    <div className="text-sm text-yellow-600">= 2 points</div>
                  </div>
                  <div className="p-3 bg-blue-50 rounded-lg border border-blue-200">
                    <div className="font-semibold text-blue-900">50-59</div>
                    <div className="text-sm text-blue-600">= 3 points</div>
                  </div>
                  <div className="p-3 bg-green-50 rounded-lg border border-green-200">
                    <div className="font-semibold text-green-900">60-69</div>
                    <div className="text-sm text-green-600">= 4 points</div>
                  </div>
                  <div className="p-3 bg-emerald-50 rounded-lg border border-emerald-200">
                    <div className="font-semibold text-emerald-900">70-100</div>
                    <div className="text-sm text-emerald-600">= 5 points</div>
                  </div>
                </div>
              </CardContent>
            </Card>
          </>
        )}
      </div>
    </div>
  )
}
